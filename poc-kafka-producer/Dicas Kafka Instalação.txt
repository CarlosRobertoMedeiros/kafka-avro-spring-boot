	log.dirs=C:/Users/carlo/Downloads/kafka_2.13-3.2.0/kafka-logs

	dataDir=C:/Users/carlo/Downloads/kafka_2.13-3.2.0/zookeper-data

    - Inicialize o zookeper e o server
         entre no diretório: C:\Users\carlo\Downloads\kafka_2.13-3.2.0\kafka_2.13-3.2.0\bin\windows
	execute: bin\windows\zookeeper-server-start.bat config\zookeeper.properties

    - Inicialize o server (Se tiver mais de um servidor tem que levantar uma instância para cada)
	entre no diretório: C:\Users\carlo\Downloads\kafka_2.13-3.2.0\
	execute: bin\windows\kafka-server-start.bat config\server.properties

    - Criação de Tópico solicitacao-transferencia

	bin\windows\kafka-topics.bat --create --bootstrap-server localhost:9092 --replication-factor 1 --partitions 1 --topic solicitacao-transferencia
	
     -  Listar os tópicos

	bin\windows\kafka-topics.bat --list --bootstrap-server localhost:9092
	
      - Criando um produtor
	kafka-console-producer.bat --broker-list localhost:9092 --topic LOJA_NOVO_PEDIDO
       
      - Criando um consumidor
	kafka-console-consumer.bat --bootstrap-server localhost:9092 --topic LOJA_NOVO_PEDIDO //A partir de agora
